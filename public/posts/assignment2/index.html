import java.util.*;

public class Rotate {
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int r = sc.nextInt(); int c = sc.nextInt(); sc.nextLine();
		int[][] input = new int[r][c];
		int rcount = 0;
		while (sc.hasNextLine() && rcount < r) {
			String temp = sc.nextLine();
			String[] values = temp.split(" ");
			for (int i = 0; i < c; i++) {
				input[rcount][i] = Integer.parseInt(values[i]);
			}
			++rcount;
		}
		sc.close();
		int[][] output = rotate(input, c, r);

        for (int i = 0; i < c; i++) {
            for (int j = 0; j < r; j++) {
                System.out.print(output[i][j] + " ");
            }
            System.out.println();
        }
	}

	public static int[][] rotate(int[][] input, int c, int r) {
		int[][] output = new int[c][r];
		for (int m = r-1; m>-1; m--) {
			for (int n = 0; n<c; n++) {
				output[n][r-1-m] = input[m][n]; // å”¯ä¸€éš¾ç‚¹
			}
		}
		return output;
	}
}

import java.util.*;

public class Mapper {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int r = sc.nextInt();
        int c = sc.nextInt();
        int[][] chart = new int[r][c];
        
        for (int row = 0; row < r; row++) { // è¯»å–æ•°ç»„
            for (int col = 0; col < c; col++) {
                chart[row][col] = sc.nextInt();
            }
        }
        sc.close();
        
        int[][] map = map(r, c, chart);
        System.out.println("Resulting Map:");
        for (int[] row : map) { // è¾“å‡ºæ•°ç»„
            for (int val : row) {
                System.out.print(val + " ");
            }
            System.out.println();
        }
    }

    public static int[][] map(int r, int c, int[][] chart) {
        int[][] dp = new int[r][c]; // ç”¨äºå­˜å‚¨åˆ°è¾¾æ¯ä¸ªç‚¹çš„æœ€å°è·¯å¾„å’Œ
        int[][] path = new int[r][c]; // ç”¨äºå­˜å‚¨è·¯å¾„å›¾
        dp[0][0] = chart[0][0];
        
        // å¡«å……ç¬¬ä¸€è¡Œå’Œç¬¬ä¸€åˆ—
        for (int i = 1; i < r; i++) {
            dp[i][0] = dp[i - 1][0] + chart[i][0];
        }
        for (int j = 1; j < c; j++) {
            dp[0][j] = dp[0][j - 1] + chart[0][j];
        }

        // è®¡ç®—å…¶ä»–ä½ç½®çš„æœ€å°è·¯å¾„å’Œ
        for (int i = 1; i < r; i++) {
            for (int j = 1; j < c; j++) {
                dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1]) + chart[i][j];
            }
        }

        // ä»å³ä¸‹è§’å›æº¯åˆ°å·¦ä¸Šè§’ï¼Œæ ‡è®°è·¯å¾„
        int x = r - 1, y = c - 1;
        path[x][y] = 1;
        
        while (x > 0 || y > 0) {
            if (x == 0) { // æœ€ä¸Šä¸€è¡Œï¼Œåªèƒ½ä»å·¦è¾¹æ¥
                y--;

                x--;
            } else { // ä»å·¦è¾¹æ¥
                y--;
            }
            path[x][y] = 1;
        }
        return path;
    }
